/**
 * @param {number[]} nums
 * @return {number}
 */
var minStartValue = function(nums) {
    // sum of prefix and find the lowest
    let minSum = 0
    let preSum = 0
    const len = nums.length

    for(let i = 0; i < len; i++){
        preSum += nums[i]
        if(preSum < minSum ) minSum = preSum
    }
    return  1 - minSum
};
